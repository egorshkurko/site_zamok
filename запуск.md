# Установка

cd /root/site_zamok_git

# Отменить локальные изменения (если не нужны)
git reset --hard HEAD

# Получить обновления с GitHub
git pull origin main

# Пересоздать venv с правильным Python
rm -rf venv
python3 -m venv venv
source venv/bin/activate
pip install --upgrade pip
pip install -r requirements.txt
deactivate
sudo systemctl restart gunicorn

# бэкап бд в 3 утра
0 3 * * * /root/site_zamok_git/backup_db.sh >> /var/log/db_backup.log 2>&1

# права на статику
chmod 755 /root
chmod 755 /root/site_zamok_git
chmod -R 755 staticfiles/
chmod -R 755 media/



# После изменений
source venv/bin/activate
python manage.py migrate
sudo systemctl restart gunicorn

## 1. Логи Gunicorn

sudo journalctl -u gunicorn -n 100 --no-pager
